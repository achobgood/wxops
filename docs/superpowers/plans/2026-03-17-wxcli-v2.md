## Shared Pattern

Every command file follows this exact structure. Subagents: read `src/wxcli/commands/locations.py` first — it is your template.

```python
import typer
from wxc_sdk.rest import RestError   # only if needed for error handling
from wxcli.auth import get_api
from wxcli.output import print_table, print_json

app = typer.Typer(help="Description.")

@app.command("list")
def list_things(
    location_id: str = typer.Option(None, "--location", help="Filter by location"),
    output: str = typer.Option("table", "--output", "-o", help="Output format: table|json"),
    limit: int = typer.Option(50, "--limit", help="Max results (0 for all)"),
    debug: bool = typer.Option(False, "--debug"),
):
    """List things."""
    api = get_api(debug=debug)
    items = list(api.telephony.FEATURE.list(location_id=location_id))
    if output == "json":
        print_json(items)
    else:
        print_table(items, columns=[...], limit=limit)

@app.command("show")
def show_thing(
    location_id: str = typer.Option(..., "--location"),
    thing_id: str = typer.Argument(help="Thing ID"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Show details."""
    api = get_api(debug=debug)
    thing = api.telephony.FEATURE.details(location_id=location_id, thing_id=thing_id)
    print_json(thing)

# create, update, delete follow same pattern
```

**IMPORTANT RULES for all subagents:**
1. Do NOT put `#` comments inside multi-line `python3.11 -c` strings — permission system blocks them
2. Use named delete methods, NOT generic `.delete()` — see spec for exact method names
3. Always use keyword arguments for SDK calls (prevents parameter order bugs)
4. `list()` returns generators — wrap in `list()` before passing to `print_table`
5. For `--output json`, use `print_json()`. For table, use `print_table()` with columns list.
6. `columns` uses dot-notation for nested fields: `("City", "address.city")`

```python
import typer
from wxc_sdk.common.schedules import Schedule, Event, ScheduleType
from wxcli.auth import get_api
from wxcli.output import print_table, print_json

app = typer.Typer(help="Manage schedules (business hours and holidays).")

@app.command("list")
def list_schedules(
    location_id: str = typer.Option(..., "--location", help="Location ID"),
    schedule_type: str = typer.Option(None, "--type", help="Filter: businessHours|holidays"),
    output: str = typer.Option("table", "--output", "-o"),
    limit: int = typer.Option(50, "--limit"),
    debug: bool = typer.Option(False, "--debug"),
):
    """List schedules for a location."""
    api = get_api(debug=debug)
    kwargs = {}
    if schedule_type:
        kwargs["schedule_type"] = ScheduleType(schedule_type)
    items = list(api.telephony.schedules.list(obj_id=location_id, **kwargs))
    if output == "json":
        print_json(items)
    else:
        print_table(items, columns=[
            ("ID", "schedule_id"),
            ("Name", "name"),
            ("Type", "schedule_type"),
        ], limit=limit)

@app.command("show")
def show_schedule(
    location_id: str = typer.Option(..., "--location"),
    schedule_type: str = typer.Option(..., "--type", help="businessHours|holidays"),
    schedule_id: str = typer.Argument(help="Schedule ID"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Show schedule details."""
    api = get_api(debug=debug)
    schedule = api.telephony.schedules.details(
        obj_id=location_id,
        schedule_type=ScheduleType(schedule_type),
        schedule_id=schedule_id,
    )
    print_json(schedule)

@app.command("create")
def create_schedule(
    location_id: str = typer.Option(..., "--location"),
    name: str = typer.Option(..., "--name"),
    schedule_type: str = typer.Option(..., "--type", help="businessHours|holidays"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Create a new schedule."""
    api = get_api(debug=debug)
    schedule = Schedule(name=name, schedule_type=ScheduleType(schedule_type))
    result = api.telephony.schedules.create(obj_id=location_id, schedule=schedule)
    typer.echo(f"Created: {result} ({name})")

@app.command("update")
def update_schedule(
    location_id: str = typer.Option(..., "--location"),
    schedule_type: str = typer.Option(..., "--type"),
    schedule_id: str = typer.Argument(help="Schedule ID"),
    name: str = typer.Option(None, "--name"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Update a schedule."""
    api = get_api(debug=debug)
    schedule = api.telephony.schedules.details(
        obj_id=location_id, schedule_type=ScheduleType(schedule_type), schedule_id=schedule_id
    )
    if name:
        schedule.new_name = name
    api.telephony.schedules.update(
        obj_id=location_id, schedule=schedule,
        schedule_type=ScheduleType(schedule_type), schedule_id=schedule_id
    )
    typer.echo(f"Updated: {schedule_id}")

@app.command("delete")
def delete_schedule(
    location_id: str = typer.Option(..., "--location"),
    schedule_type: str = typer.Option(..., "--type"),
    schedule_id: str = typer.Argument(help="Schedule ID"),
    force: bool = typer.Option(False, "--force"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Delete a schedule."""
    if not force:
        typer.confirm(f"Delete schedule {schedule_id}?", abort=True)
    api = get_api(debug=debug)
    api.telephony.schedules.delete_schedule(
        obj_id=location_id, schedule_type=ScheduleType(schedule_type), schedule_id=schedule_id
    )
    typer.echo(f"Deleted: {schedule_id}")

@app.command("add-event")
def add_event(
    location_id: str = typer.Option(..., "--location"),
    schedule_type: str = typer.Option(..., "--type"),
    schedule_id: str = typer.Option(..., "--schedule-id"),
    name: str = typer.Option(..., "--name"),
    start_date: str = typer.Option(..., "--start-date", help="YYYY-MM-DD"),
    end_date: str = typer.Option(..., "--end-date", help="YYYY-MM-DD"),
    all_day: bool = typer.Option(True, "--all-day/--no-all-day"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Add an event to a schedule."""
    import datetime
    api = get_api(debug=debug)
    event = Event(
        name=name,
        start_date=datetime.date.fromisoformat(start_date),
        end_date=datetime.date.fromisoformat(end_date),
        all_day_enabled=all_day,
    )
    result = api.telephony.schedules.event_create(
        obj_id=location_id, schedule_type=ScheduleType(schedule_type),
        schedule_id=schedule_id, event=event,
    )
    typer.echo(f"Created event: {result} ({name})")

@app.command("show-event")
def show_event(
    location_id: str = typer.Option(..., "--location"),
    schedule_type: str = typer.Option(..., "--type"),
    schedule_id: str = typer.Option(..., "--schedule-id"),
    event_id: str = typer.Argument(help="Event ID"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Show event details."""
    api = get_api(debug=debug)
    event = api.telephony.schedules.event_details(
        obj_id=location_id, schedule_type=ScheduleType(schedule_type),
        schedule_id=schedule_id, event_id=event_id,
    )
    print_json(event)

@app.command("delete-event")
def delete_event(
    location_id: str = typer.Option(..., "--location"),
    schedule_type: str = typer.Option(..., "--type"),
    schedule_id: str = typer.Option(..., "--schedule-id"),
    event_id: str = typer.Argument(help="Event ID"),
    force: bool = typer.Option(False, "--force"),
    debug: bool = typer.Option(False, "--debug"),
):
    """Delete an event from a schedule."""
    if not force:
        typer.confirm(f"Delete event {event_id}?", abort=True)
    api = get_api(debug=debug)
    api.telephony.schedules.event_delete(
        obj_id=location_id, schedule_type=ScheduleType(schedule_type),
        schedule_id=schedule_id, event_id=event_id,
    )
    typer.echo(f"Deleted event: {event_id}")
```
